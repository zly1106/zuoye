def bubblesort(s):
    a = -1
    while True:
        a = a + 1
        if a > len(s):
            break
        for i in range(a, len(s)):
            if s[i] > s[a] and a < len(s):
                s[i], s[a] = s[a], s[i]
l = [2,2,9,10,3]
bubblesort(l)
print(l)
